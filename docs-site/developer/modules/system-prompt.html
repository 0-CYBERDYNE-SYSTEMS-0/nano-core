system-prompt.ts

Builds the full system prompt for each pi run: base layer caching, context file injection, skill catalog, capability routing, budget management, and injection detection.

Types

type CodingHint = 'none' | 'auto' | 'force_delegate_execute' | 'force_delegate_plan'
type PromptMode = 'full' | 'minimal'

interface SystemPromptInput {
  groupFolder: string; chatJid: string; isMain: boolean
  isScheduledTask?: boolean; isHeartbeatTask?: boolean
  assistantName?: string; provider?: string; model?: string
  thinkLevel?: ThinkLevel; reasoningLevel?: ReasoningLevel
  noContinue?: boolean; memoryContext?: string
  codingHint: CodingHint; requestId?: string
  extraSystemPrompt?: string; skillCatalog?: SkillCatalogEntry[]
}

interface WorkspacePaths {
  groupDir: string; globalDir: string; ipcDir: string
}

interface SystemPromptReport {
  mode: PromptMode; totalChars: number
  contextEntries: ContextEntry[]   // per file: label, path, chars, truncated, missing, blocked
  contextBudget: { fileMaxChars: number; totalMaxChars: number; injectedTotalChars: number; remainingChars: number }
  layers: PromptLayer[]            // base + overlays
  baseCacheKey: string; basePromptHash: string; cacheHit: boolean
  skillsCatalog: { count: number; injectedChars: number; truncated: boolean }
}

Main Function

buildSystemPrompt(
  input: SystemPromptInput,
  paths: WorkspacePaths,
  options?: {
    delegationExtensionAvailable?: boolean
    readFileIfExists?: (filePath: string) => string | null
    now?: () => Date
    fileMaxChars?: number          // default from env
    totalMaxChars?: number         // default from env
    skillCatalogMaxChars?: number  // default from env
    cachedBaseLayer?: { key: string; hash: string; content: string } | null
  },
): { text: string; report: SystemPromptReport }

The base layer (assistant identity, workspace rules, capability routing) is cacheable by key+hash. Context files (SOUL.md, MEMORY.md, NANO.md, etc.) are injected within budget. Prompt injection patterns in file content are detected and redacted.
